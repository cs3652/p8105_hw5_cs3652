for (i in 1:20)
  if (i <= 10) {
    study_df[[i]] = study_df[[i]] %>% 
      mutate(arm = "control", study_id = i)
  } else if (i > 10) {
    study_df[[i]] = study_df[[i]] %>% 
      mutate(arm = "experimental", study_id = i - 10)
  }

study_data = bind_rows(study_df) %>%
